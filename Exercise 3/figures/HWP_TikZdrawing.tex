    \draw[axis] (0,-1.5) -- (0,1.5) node[above] {Slow Axis ($s$)};

    \coordinate (O) at (0,0);

    % Components of E_in
    \draw[component] (0,0) -- ({1.3*cos(30)}, 0) node[below] {$E_f$};
    \draw[component] (0,0) -- (0, {1.3*sin(30)}) node[left] {$E_s$};

    \node[draw, fill=white, font=\small, text width=4cm, align=left] at (1.8, 1) {
        \textbf{Phase Retardance}: \\
        $E_f \rightarrow E_f$ \\
        $E_s \rightarrow E_s \cdot e^{i\pi} = -E_s$ \\
        The slow component is delayed by $\lambda/2$.
    };

    % Resulting Vector (E_out)
