
\begin{tikzpicture}[scale=2, >=Stealth, 
    axis/.style={dashed, ->, gray, thin},
    vector/.style={very thick, ->},
    component/.style={thick, blue, ->}]

    % Coordinate Axes (Slow and Fast)
    \draw[axis] (-1,0) -- (1.2,0) node[right] {Slow Axis ($s$)};
    \draw[axis] (0,-0.15) -- (0,1.2) node[above] {Fast Axis ($f$)};

    \coordinate (Ein) at (50:1.3);
    \coordinate (Eout) at (40+90:1.3);

    \draw[loosely dotted, gray] (Eout) -- (Ein);
    \draw[loosely dotted, gray] (-{1.3*sin(40)}, 0) -- (Eout);
    \draw[loosely dotted, gray] ({1.3*sin(40)}, 0) -- (Ein);

    % Incident Light Vector (E_in)
     \coordinate (O) at (0,0);
    \draw[vector, red] (O) -- (Ein) node[above left] {$\mathbf{E}_{in}$};
    
    % Components of E_in
    \draw[component] (0,0) -- (0, {1.3*cos(40)}) node[anchor=north east] {$E_f$};
    \draw[component] (0,0) -- ({1.3*sin(40)}, 0) node[anchor=south east] {$E_s$};

    % Resulting Vector (E_out)
    \draw[vector, green!70!black] (O) -- (Eout) node[above right] {$\mathbf{E}_{out}$};
    
    % Components of E_out
    \draw[component, dashed] (0,0) -- ({-1.3*sin(40)}, 0) node[anchor=south west] {$-E_s$};

    \pic [draw, ->,
      angle radius=7mm, angle eccentricity=1.2] {angle = Ein--O--Eout};
      \node at (75:0.45) {$\theta$};
\end{tikzpicture}
